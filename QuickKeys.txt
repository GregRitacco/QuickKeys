Proc HelpUtilList ** Pass in xMatrix

Let xHelpAll = ""
Let xUtilAll = ""
Let i = 0
While (i < $Set(%1, Count))
  If ($Set(%1, Get, @i) = 1)
    Let xHelpAll = @xHelpAll $Set(@xHelpSet, Get, @i)
    Let xUtilAll = @xUtilAll + $Set(@xUtilSet, Get, @i)
  EndIf
  Let i = @i + 1
EndWhile
Let xUtilAll = @xUtilAll + \n\n$Set(@xUtilSet, Get, 9) \n

EndProc
